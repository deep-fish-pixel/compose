资源引用(五)

支持对象索引和amd组合使用

    compose.require(['mod/page/first-page'], function(pfirst){
        //该文件入口
        alert(pfirst.id);
    }, ['pfirst']);
    
    compose.require(['mod/page/second-page'], function(wthird){
        //该文件入口
        alert(wthird.id);
    });
    
    compose.require(['mod/a', 'mod/b'], function(a, b){
        //该文件入口
        alert(a);
        alert(b);
    }, ['a', 'b']);
    compose.require(['mod/k'], function(k){
        //该文件入口
        alert(k.id);
    }, ['k']);
